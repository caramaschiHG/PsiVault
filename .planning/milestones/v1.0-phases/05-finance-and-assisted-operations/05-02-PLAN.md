```typescript
export type ChargeStatus = "pendente" | "pago" | "atrasado";
export type PaymentMethod = "pix" | "transferencia" | "dinheiro" | "outro" | null;

export interface SessionCharge {
  id: string;
  workspaceId: string;
  patientId: string;
  appointmentId: string;
  status: ChargeStatus;
  amountInCents: number | null;
  paymentMethod: PaymentMethod;
  createdAt: Date;
  updatedAt: Date;
  paidAt: Date | null;
}

export function createSessionCharge(
  input: { workspaceId: string; patientId: string; appointmentId: string; amountInCents: number | null },
  deps: { now: Date; createId: () => string },
): SessionCharge;

export function updateSessionCharge(
  charge: SessionCharge,
  input: { status?: ChargeStatus; amountInCents?: number | null; paymentMethod?: PaymentMethod },
  deps: { now: Date },
): SessionCharge;

export function deriveFinancialStatus(charges: SessionCharge[]): FinancialStatus;
export function deriveMonthlyFinancialSummary(charges: SessionCharge[]): MonthlyFinancialSummary;
```
