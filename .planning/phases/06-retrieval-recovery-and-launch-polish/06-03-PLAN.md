export type SearchDomain = "patient" | "session" | "document" | "charge";

export interface SearchResultItem {
  id: string;
  domain: SearchDomain;
  patientName: string;
  label: string;       // e.g. "Paciente", "Sessão — 14 jan 2026", "Declaração de comparecimento", "Cobrança pendente"
  href: string;        // navigation target
  date?: string;       // ISO date string — present for sessions, documents, charges
  // SECU-05: NO freeText, amountInCents, demand, observedMood, or document body fields
}

export interface SearchResults {
  patients: SearchResultItem[];    // up to 3 shown in dropdown
  sessions: SearchResultItem[];
  documents: SearchResultItem[];
  charges: SearchResultItem[];
}

export interface SearchInput {
  query: string;
  patients: Patient[];
  appointments: Appointment[];
  documents: Document[];
  charges: SessionCharge[];
  patientMap: Map<string, Patient>;
}

export function searchAll(input: SearchInput): SearchResults;
// Each array limited to 10 items max (dropdown shows 3, "Ver todos" links to full section)
// Case-insensitive substring match: value.toLowerCase().includes(query.toLowerCase())
// Empty query returns { patients: [], sessions: [], documents: [], charges: [] }
