7. **Update** load/save to use storage:
```typescript
// Load from storage on mount
useEffect(() => {
  if (!initialized.current) {
    initialized.current = true;
    setNotifications(storage.load());
  }
}, [storage]);

// Persist to storage on change
useEffect(() => {
  if (initialized.current) {
    storage.save(notifications);
  }
}, [notifications, storage]);
```

8. **Update** addNotification to use CreateNotificationInput:
```typescript
const addNotification = (n: CreateNotificationInput) => {
  const newNotif: AppNotification = {
    ...n,
    id: `notif-${Date.now()}-${Math.random().toString(36).slice(2, 7)}`,
    read: false,
    createdAt: Date.now(),
  };
  setNotifications((prev) => [newNotif, ...prev].slice(0, 50));
};
```

9. **Add** removeNotification:
```typescript
const removeNotification = (id: string) => {
  setNotifications((prev) => prev.filter((n) => n.id !== id));
};
```

10. **Update** clearAll to use storage:
```typescript
const clearAll = () => {
  setNotifications([]);
  storage.clear();
};
```

11. **Re-export** AppNotification type for backward compatibility:


12. Keep `useNotifications` hook unchanged.
