# Plan 01 — Schema Prisma + Domínio de Categorias

## Objective

Estabelecer a fundação do módulo: schema Prisma para `Expense` e `ExpenseCategory`,
aplicar via `prisma db push`, e construir o domínio de categorias seguindo o
repository pattern do projeto. Esta é a Wave 0 — todas as outras plans dependem
de o schema estar aplicado.

**Action:**

Adicionar ao final de `prisma/schema.prisma`:

```prisma
model ExpenseCategory {
  id          String   @id
  workspaceId String   @map("workspace_id")
  name        String
  archived    Boolean  @default(false)
  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")
  deletedAt   DateTime? @map("deleted_at")
  deletedByAccountId String? @map("deleted_by_account_id")

  expenses    Expense[]

  @@index([workspaceId, deletedAt])
  @@index([workspaceId, archived])
  @@map("expense_categories")
}

model Expense {
  id                  String   @id
  workspaceId         String   @map("workspace_id")
  categoryId          String   @map("category_id")
  description         String
  amountInCents       Int      @map("amount_in_cents")
  dueDate             DateTime @map("due_date")
  recurrencePattern   String?  @map("recurrence_pattern") // "MENSAL" | "QUINZENAL"
  seriesId            String?  @map("series_id")
  seriesIndex         Int?     @map("series_index")
  receiptStorageKey   String?  @map("receipt_storage_key")
  receiptFileName     String?  @map("receipt_file_name")
  receiptMimeType     String?  @map("receipt_mime_type")
  createdAt           DateTime @default(now()) @map("created_at")
  updatedAt           DateTime @updatedAt @map("updated_at")
  createdByAccountId  String   @map("created_by_account_id")
  deletedAt           DateTime? @map("deleted_at")
  deletedByAccountId  String?  @map("deleted_by_account_id")

  category ExpenseCategory @relation(fields: [categoryId], references: [id], onDelete: Restrict)

  @@index([workspaceId, deletedAt])
  @@index([workspaceId, dueDate])
  @@index([seriesId])
  @@map("expenses")
}
```

**Acceptance criteria:**

- `rg "model ExpenseCategory" prisma/schema.prisma` retorna 1 match
- `rg "@@index\(\[workspaceId, deletedAt\]\)" prisma/schema.prisma` retorna ≥2 matches (uma para cada model)

**Action:**

1. `model.ts`:
   ```ts
   export interface ExpenseCategory {
     id: string;
     workspaceId: string;
     name: string;
     archived: boolean;
     createdAt: Date;
     updatedAt: Date;
     deletedAt: Date | null;
     deletedByAccountId: string | null;
   }

   export interface CreateExpenseCategoryInput {
     workspaceId: string;
     name: string;
   }

   export interface CreateExpenseCategoryDeps {
     now: Date;
     createId: () => string;
   }

   export function createExpenseCategory(
     input: CreateExpenseCategoryInput,
     deps: CreateExpenseCategoryDeps,
   ): ExpenseCategory {
     return {
       id: deps.createId(),
       workspaceId: input.workspaceId,
       name: input.name.trim(),
       archived: false,
       createdAt: deps.now,
       updatedAt: deps.now,
       deletedAt: null,
       deletedByAccountId: null,
     };
   }

   export function renameExpenseCategory(
     category: ExpenseCategory,
     newName: string,
     deps: { now: Date },
   ): ExpenseCategory {
     return { ...category, name: newName.trim(), updatedAt: deps.now };
   }

   export function archiveExpenseCategory(
     category: ExpenseCategory,
     deps: { now: Date },
   ): ExpenseCategory {
     return { ...category, archived: true, updatedAt: deps.now };
   }
   ```
