```tsx
function CurrentTimeIndicator({ dayStartHour, pixelsPerMinute }: { ... }) {
  const [now, setNow] = useState(new Date());

  useEffect(() => {
    const interval = setInterval(() => setNow(new Date()), 60_000);
    return () => clearInterval(interval);
  }, []);

  const minutesSinceStart = (now.getUTCHours() * 60 + now.getUTCMinutes()) - (dayStartHour * 60);
  const topPx = minutesSinceStart * pixelsPerMinute;

  // Só renderizar se dentro do viewport do grid
  if (topPx < 0 || topPx > containerHeight) return null;

  return (
    <div style={{
      position: "absolute",
      top: topPx,
      left: 0,
      right: 0,
      height: "2px",
      background: "#dc2626",
      zIndex: 10,
    }}>
      <div style={{
        position: "absolute",
        left: 0,
        top: "-3px",
        width: "8px",
        height: "8px",
        borderRadius: "50%",
        background: "#dc2626",
      }} />
    </div>
  );
}
```
