<task type="auto">
  <name>Task 2: Create src/styles/motion.css with keyframes, utilities, and reduced-motion</name>
  <files>src/styles/motion.css</files>
  <action>
    Create the directory `src/styles/` and file `src/styles/motion.css` with the following content, organized in sections:

    ```css
    /* ─── Motion Utilities & Keyframes — v1.5 Foundation ─────────────────────── */

    /* ═══════════════════════════════════════════════════════════════
       KEYFRAMES — Canonical enter/exit animations
       ═══════════════════════════════════════════════════════════════ */

    @keyframes fadeIn {
      from { opacity: 0; }
      to   { opacity: 1; }
    }

    @keyframes fadeOut {
      from { opacity: 1; }
      to   { opacity: 0; }
    }

    @keyframes slideUp {
      from { opacity: 0; transform: translateY(8px); }
      to   { opacity: 1; transform: translateY(0); }
    }

    @keyframes slideDown {
      from { opacity: 0; transform: translateY(-8px); }
      to   { opacity: 1; transform: translateY(0); }
    }

    @keyframes scaleIn {
      from { opacity: 0; transform: scale(0.95); }
      to   { opacity: 1; transform: scale(1); }
    }

    /* ═══════════════════════════════════════════════════════════════
       UTILITY CLASSES — Server Component safe, no JS required
       ═══════════════════════════════════════════════════════════════ */

    .motion-fade-in {
      animation: fadeIn var(--duration-200) var(--ease-out) both;
    }

    .motion-fade-out {
      animation: fadeOut var(--duration-200) var(--ease-out) both;
    }

    .motion-slide-up {
      animation: slideUp var(--duration-200) var(--ease-out) both;
    }

    .motion-slide-down {
      animation: slideDown var(--duration-200) var(--ease-out) both;
    }

    .motion-scale-in {
      animation: scaleIn var(--duration-200) var(--ease-out) both;
    }

    .motion-stagger > * {
      animation: fadeIn var(--duration-200) var(--ease-out) both;
      animation-delay: calc(var(--stagger-index, 0) * var(--stagger-gap));
    }

    /* ═══════════════════════════════════════════════════════════════
       REDUCED MOTION — Respect user preference
       ═══════════════════════════════════════════════════════════════ */

    @media (prefers-reduced-motion: reduce) {
      .motion-fade-in,
      .motion-fade-out,
      .motion-slide-up,
      .motion-slide-down,
      .motion-scale-in,
      .motion-stagger > * {
        animation: none !important;
        opacity: 1 !important;
        transform: none !important;
      }
    }
    ```

    The file must NOT include the existing app-specific keyframes (vaultPageIn, toastSlideIn, etc.) — those stay in globals.css for now and are migrated in Plan 02.
