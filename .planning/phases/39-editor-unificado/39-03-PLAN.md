       ```typescript
       "use server";
       
       export async function signDocumentAction(formData: FormData): Promise<void> {
         const { workspaceId, accountId } = await resolveSession();
         const documentId = formData.get("documentId") as string;
         const patientId = formData.get("patientId") as string;
         
         const docRepo = getDocumentRepository();
         const doc = await docRepo.findById(documentId, workspaceId);
         if (!doc || doc.patientId !== patientId) {
           throw new Error("Documento não encontrado.");
         }
         
         if (!canSignDocument(doc)) {
           throw new Error("Este documento não pode ser assinado.");
         }
         
         const profile = await getPracticeProfileSnapshot(undefined, workspaceId);
         if (!profile.signatureAsset?.storageKey) {
           throw new Error("Configure sua assinatura em Perfil para finalizar este documento.");
         }
         
         const updated = signDocument(doc, accountId, { now: new Date() });
         await docRepo.update(documentId, workspaceId, {
           status: updated.status,
           signedAt: updated.signedAt,
           signedByAccountId: updated.signedByAccountId,
         });
         
         revalidatePath(`/patients/${patientId}/documents/${documentId}`, "page");
         revalidatePath(`/patients/${patientId}`, "page");
       }
       ```
