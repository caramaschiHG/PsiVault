       ```typescript
       import { isSameDay } from "date-fns";
       import type { Appointment } from "../../appointments/model";

       export interface DailySummaryInput {
         date: Date;
         workspaceId: string;
         accountId: string; // the psychologist who scheduled the appointments
         noShowsDetected: number;
         remindersSent: number;
         suggestionsGenerated: number;
       }

       export interface DailySummaryOutput {
         title: string;
         description: string;
         dateLabel: string;
       }

       export function generateDailySummary(input: DailySummaryInput): DailySummaryOutput {
         const dateLabel = input.date.toLocaleDateString("pt-BR", {
           day: "numeric",
           month: "long",
           timeZone: "America/Sao_Paulo",
         });
         const parts: string[] = [];
         if (input.noShowsDetected > 0) {
           parts.push(`${input.noShowsDetected} falta${input.noShowsDetected > 1 ? "s" : ""} detectada${input.noShowsDetected > 1 ? "s" : ""}`);
         }
         if (input.remindersSent > 0) {
           parts.push(`${input.remindersSent} lembrete${input.remindersSent > 1 ? "s" : ""} enviado${input.remindersSent > 1 ? "s" : ""}`);
         }
         if (input.suggestionsGenerated > 0) {
           parts.push(`${input.suggestionsGenerated} sugestão${input.suggestionsGenerated > 1 ? "s" : ""} de horário gerada${input.suggestionsGenerated > 1 ? "s" : ""}`);
         }
         const description = parts.length > 0 ? parts.join(" · ") : "Nenhuma atividade da agenda hoje.";
         return {
           title: `Resumo do dia — ${dateLabel}`,
           description,
           dateLabel,
         };
       }

       export function isLastAppointmentOfDay(
         completedAppointment: Appointment,
         allTodayAppointments: Appointment[],
       ): boolean {
         const today = completedAppointment.startsAt;
         const laterToday = allTodayAppointments.filter(
           (a) =>
             isSameDay(a.startsAt, today) &&
             a.startsAt > completedAppointment.startsAt &&
             (a.status === "SCHEDULED" || a.status === "CONFIRMED"),
         );
         return laterToday.length === 0;
       }

       export function buildDailySummaryIdempotencyKey(
         workspaceId: string,
         accountId: string,
         date: Date,
       ): string {
         const dateKey = date.toISOString().slice(0, 10);
         return `agenda:daily-summary:${workspaceId}:${accountId}:${dateKey}`;
       }
       ```

       The `"daily_summary"` case in `executeTask` (already updated in Task 1) should also **create the notification** and return it in the output so the cron handler or action can inject it into the client's notification storage.
       Actually, the agent task runs on the server (cron processor). The notification is client-side (localStorage). The bridge:
       - Option A: The agent task returns the summary data, and the cron handler's audit loop stores it in a server-side location that the client polls.
       - Option B: The appointment action itself creates the notification client-side via a Server Action that returns the notification payload, which the client then saves to localStorage.
       - **Recommended (MVP):** The `completeAppointmentAction` returns the notification payload to the client when `isLast` is true. The client-side mutation handler (`useMutation` or form action response) receives the payload and calls `addNotification()` from the notification context.

       - This avoids server→client push complexity and respects the existing localStorage architecture.

       However, the agent task itself also runs via cron. For the cron path, the summary would only be visible on the next page load or when the client polls. That's acceptable for a low-priority daily digest (D-25).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| appointment action → notification | Summary is computed server-side and passed to client; client cannot forge summary without completing the last appointment |
| localStorage → notification display | 7-day auto-cleanup prevents unbounded growth |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-44-01 | Spoofing | Client forges `agent_summary` notification | mitigate | Summary payload is generated server-side in `completeAppointmentAction`; client only receives it after valid appointment completion |
| T-44-02 | Information Disclosure | `agent_summary` contains workspace-wide counts | accept | Counts are aggregated statistics, not PII. Only the scheduling psychologist sees the summary. |
| T-44-03 | Denial of Service | localStorage quota exceeded by many summaries | mitigate | 7-day auto-cleanup + 50-item MAX_STORED limit in `LocalNotificationStorage` |
| T-44-04 | Tampering | Duplicate summaries for same day | mitigate | Idempotency key `agenda:daily-summary:{workspaceId}:{accountId}:{date}` prevents duplicate agent tasks |
</threat_model>
